    #include <iostream>
    #include <stdlib.h>
    #define size 20
    using namespace std;
    class stack
    {
     int a[size],top,ms;
     public:
      stack()
      {
               top=-1; cout<<"MAX:";cin>>ms;
     }

     stack operator +(int ele)
     {
       if(top>ms-2)
       cout<<"Stack Overflow\n";
       else
           {
                top++;
                a[top]=ele;
            }
       
           return (*this);
     }
    
      stack operator --(int)
      {
        if(top==-1)
            cout<<"Stack Underflow";
        else
        cout<<"\nElement deleted is "<<a[top--];
        return (*this);
     }

     friend ostream& operator <<(ostream& c,stack s)
     {
        if(s.top==-1)
              cout<<"Stack Empty";
        else
        for(int j=s.top;j>=0;j--)
                     cout<<" "<<s.a[j];
        return c;
     }
    }s;
